#! /bin/sh
echo "$(tput setaf 0)$(tput setab 4)install SSR$(tput sgr0)"
#ask pwd
echo "Please enter your SSR password";
stty -echo
read mypwd;
stty echo
#install ssr
apt -y update >/dev/null 2>&1
apt -y install git >/dev/null 2>&1
git clone --quiet https://github.com/ali50m/ssr.git /ssr
cd /ssr && sh ./initcfg.sh
cat > /ssr/user-config.json << 'SSREOF'
{
    "server": "0.0.0.0",
    "server_ipv6": "::",
    "server_port": 8079,
    "local_address": "127.0.0.1",
    "local_port": 1080,
    "password": "$mypwd",
    "method": "rc4-md5",
    "protocol": "auth_aes128_md5",
    "protocol_param": "",
    "obfs": "tls1.2_ticket_auth",
    "obfs_param": "",
    "speed_limit_per_con": 0,
    "speed_limit_per_user": 0,
    "additional_ports" : {}, // only works under multi-user mode
    "timeout": 120,
    "udp_timeout": 60,
    "dns_ipv6": false,
    "connect_verbose_info": 0,
    "redirect": "",
    "fast_open": false
}
SSREOF
cd ~
sh /ssr/shadowsocks/logrun.sh

# 加入开机启动
crontab -l > my-crontab >/dev/null 2>&1
if ! (grep -q '#WLIU_SSR' my-crontab) ; then
echo '
#WLIU_SSR on reboot
@reboot /ssr/shadowsocks/logrun.sh
' >> my-crontab
crontab my-crontab
update-rc.d cron enable
fi
rm my-crontab
