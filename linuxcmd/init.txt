#! /bin/sh
echo "$(tput setaf 0)$(tput setab 4)init$(tput sgr0)"
#set history format
if ! (grep -q '#WLIU_HISTORYFORMAT' ~/.bashrc) ; then
    cat >> ~/.bashrc << 'EOF'

#WLIU_HISTORYFORMAT
HISTFILESIZE=10000
HISTSIZE=10000
export HISTTIMEFORMAT="%F %T `who -u am i 2>/dev/null| awk '{print $NF}'|sed -e 's/[()]//g'` `whoami` "
EOF
fi

#set timezone
ln -sf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime

#start cron on bootup
update-rc.d cron enable

#update apt
apt -y update >/dev/null 2>&1
#apt -y install tree curl >/dev/null 2>&1

#add chs support
if ! (locale -a | grep -iq '^en_US[.]utf8$') ; then
    locale-gen en_US.utf8 > /dev/null
    if ! (set | grep 'LC_ALL=en_US[.]UTF-8') ; then
        echo '
#WLIU_CHSFONTSUPPORT
export LC_ALL=en_US.UTF-8
' >> ~/.profile
    fi
fi

# change PS1 to root@bwg:~#
echo '
#WLIU_CUSTOM_PS1
export NICKNAME=bwg
export PS1="\[\e]0;\u@$NICKNAME: \w\a\]\u@$NICKNAME:\w\\$\[$(tput sgr0)\] "' | sudo tee -a /root/.bashrc

echo "$(tput setaf 0)$(tput setab 1)please reconnect$(tput sgr0)"
#logout for enable settins
#logout
kill -HUP `pgrep -s 0 -o`
